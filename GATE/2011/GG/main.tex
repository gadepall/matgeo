\begin{multicols}{2}

	\item \(\brak{\vec{G}^\top \vec{G}}^{-1}\)  
	\item \(\vec{G}^\top \brak{\vec{G} \vec{G}^\top}^{-1}\)

\end{enumerate}
\end{multicols}
