    \item The system of equations
    \begin{align*}
        x+y+z &= 6 \\
        x+4y+6z &= 20 \\
        x+4y+\lambda z &= \mu
    \end{align*}
    has NO solution for values of $\lambda$ and $\mu$ given by
    \begin{enumerate}
        \begin{multicols}{4}
            \item $\lambda=6, \mu=20$
            \item $\lambda=6, \mu \ne 20$
            \item $\lambda \ne 6, \mu=20$
            \item $\lambda \ne 6, \mu \ne 20$
        \end{multicols}
    \end{enumerate}
