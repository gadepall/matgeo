\begin{lstlisting}[language=C, basicstyle=\ttfamily\small]
int height (treeptr n)
{
    if (n == NULL) return -1;
    if (n -> left == NULL)
        if (n -> right == NULL) return 0;
        else return B1;    // Box 1
    else {
        h1 = height(n -> left);
        if (n -> right == NULL) return (1 + h1);
        else { 
            h2 = height(n -> right);
            return B2;    // Box 2
        }
    }
}
\end{lstlisting}
