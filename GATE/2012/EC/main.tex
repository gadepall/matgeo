\item Given that
\[
\vec{A} = \begin{pmatrix}-5 & -3\\ 2 & 0\end{pmatrix}, \qquad
\vec{I} = \begin{pmatrix}1 & 0\\ 0 & 1\end{pmatrix},
\]
the value of $\vec{A}^{3}$ is
\hfill {(EC 2012)}
\begin{multicols}{4}
    \begin{enumerate}
        \item $15\vec{A} + 12\vec{I}$
        \item $19\vec{A} + 30\vec{I}$
        \item $17\vec{A} + 15\vec{I}$
        \item $17\vec{A} + 21\vec{I}$
    \end{enumerate}
\end{multicols}
