$
\myvec{
-3 & 0 & 0 & -2 \\
1 & -1 & 0 & 0 \\
0 & a & -2
}
$
has three linearly independent real eigenvectors is\hfill{{(EE 2015)}}
\begin{multicols}{4}
    \begin{enumerate}
        \item $\frac{2}{3\sqrt{3}}$
        \item $\frac{1}{3\sqrt{3}}$
        \item $\frac{1+2\sqrt{3}}{3\sqrt{3}}$
        \item $\frac{1+\sqrt{3}}{3\sqrt{3}}$
    \end{enumerate}
\end{multicols}
