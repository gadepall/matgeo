\item
What is the maximum number of free independent real parameters specifying an $n$-dimensional orthogonal matrix?
(PH 2022)
\begin{enumerate}
	\begin{multicols}{4}
    \item $n(n - 2)$
    \item $(n - 1)^2$
    \item $\frac{n(n - 1)}{2}$
    \item $\frac{n(n + 1)}{2}$
    \end{multicols}
\end{enumerate}
