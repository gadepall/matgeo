\item Let $\vec{A}$ be a square matrix of size $n \times n \brak{n > 1}$. The elements of $\vec{A} = \brak{a_{ij}}$ are given by
\begin{align*}
	a_{ij} = \begin{cases}
		i \times j, & \text{if } i \ge j \\
		0, & \text{if } i < j
	\end{cases}
\end{align*}
The determinant of $\vec{A}$ is

\end{enumerate}
