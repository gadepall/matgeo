\item
Consider the system of linear equations
\begin{align*}
x + 2y + z &= 5 \\
2x + a y + 4z &= 12 \\
2x + 4y + 6z &= b
\end{align*}
The values of $a$ and $b$ such that there exists a non-trivial null space and the system admits infinite solutions are
\hfill{\brak{\text{ME 2024}}}
\begin{enumerate}
\begin{multicols}{4}
\item $a=8, b=14$
\item $a=4, b=12$
\item $a=8, b=12$
\item $a=4, b=14$
\end{multicols}
\end{enumerate}
