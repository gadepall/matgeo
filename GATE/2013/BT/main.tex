\item 
If 
\[
\vec{P} = \begin{pmatrix}
1 & 1 \\
2 & 2
\end{pmatrix},
\quad
\vec{Q} = \begin{pmatrix}
1 & 2 \\
2 & 2
\end{pmatrix},
\quad
\vec{R} = \begin{pmatrix}
0 & 3 \\
3 & 1
\end{pmatrix}
\]
which one of the following statements is TRUE?
\hfill (BT 2013)
\begin{multicols}{4}
\begin{enumerate}
    \item $\vec{P}\vec{Q} = \vec{P}\vec{R}$
    \item $\vec{Q}\vec{R} = \vec{R}\vec{P}$
    \item $\vec{Q}\vec{P} = \vec{R}\vec{P}$
    \item $\vec{P}\vec{Q} = \vec{Q}\vec{R}$
\end{enumerate}
\end{multicols}
