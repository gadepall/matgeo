    \item A square matrix $\vec{B}$ is skew-symmetric if \hfill (CE 2009)
    \begin{multicols}{2}
    \begin{enumerate}
        \item $\vec{B}^{\top} = -\vec{B}$ 
        \item $\vec{B}^{\top} = \vec{B}$ 
        \item $\vec{B}^{-1} = \vec{B}$ 
        \item $\vec{B}^{-1} = \vec{B}^{\top}$
    \end{enumerate}
    \end{multicols}
